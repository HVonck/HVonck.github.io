#pragma once
#include < fstream >
#include "GenerationData.h"

class Menu
{
public:
	virtual std::string style(GenerationData data) { return std::string(); };
	virtual void generate(GenerationData data, std::ofstream& fout) {};
	virtual std::string javascript() { return std::string(); };
};

class DefaultMenu : public Menu
{
	virtual std::string style(GenerationData data) override;
	virtual void generate(GenerationData data, std::ofstream& fout) override;
};

class ResponsiveDefaultMenu : public Menu
{
	virtual std::string style(GenerationData data) override;
	virtual void generate(GenerationData data, std::ofstream& fout) override;
	virtual std::string javascript() override;
};

class DropdownMenu : public Menu
{
	virtual std::string style(GenerationData data) override;
	virtual void generate(GenerationData data, std::ofstream& fout) override;
};

class ResponsiveDropdownMenu : public Menu
{
	virtual std::string style(GenerationData data) override;
	virtual void generate(GenerationData data, std::ofstream& fout) override;
	virtual std::string javascript();
};
